function startGame() { stopAllSounds(); playSound(backgroundMusic); // 1. Baraja todo el banco de preguntas para obtener un orden aleatorio. const shuffledQuestions = [...FULL_QUESTION_BANK].sort(() => 0.5 - Math.random()); // 2. Reinicia y construye la nueva estructura de niveles del juego. gameLevels = []; const numberOfLevels = 8; // Se definen 8 niveles. // 3. Crea 8 niveles, cada uno con una única pregunta extraída de la lista barajada. for (let i = 0; i < numberOfLevels; i++) { // Asegurarse de que todavía hay preguntas disponibles. if (shuffledQuestions.length > 0) { gameLevels.push({ level: i + 1, // splice(0, 1) extrae el primer elemento y lo devuelve en un array, // asegurando que no se repitan preguntas y manteniendo el formato requerido. questions: shuffledQuestions.splice(0, 1) }); } } // 4. Reinicia las variables de estado del juego. currentLevel = 0; lives = 3; totalScore = 0; selectedAnswer = false; lastQuestionObject = null; // 5. Crea dinámicamente la escalera de premios para los 8 niveles. // Esta parte se adapta automáticamente a la nueva cantidad de niveles. prizeList.innerHTML = ''; gameLevels.forEach((levelData) => { const li = document.createElement('li'); li.id = `level-${levelData.level - 1}`; // Se ajusta el texto para mayor claridad, ya que cada nivel es una pregunta. li.innerHTML = `${levelData.level} PREGUNTA ${levelData.level}`; prizeList.appendChild(li); }); // 6. Cambia a la pantalla de juego y carga la primera pregunta. switchScreen(gameScreenWrapper); loadQuestion(); }
